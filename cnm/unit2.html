    double x0, x1, tol;

    {

    }
    double f_prime(double x) {
    return 3 * x * x - 1;
    }
    printf("Enter the value  x0: ");

    printf("Enter the tolerance: ");
    scanf("%lf", &tol);

    printf("Iteration\t x0\t\t x1\t\t f(x1)\n");

        double f_x0 = f(x0);
        double f_prime_x0 = f_prime(x0);

        if (f_prime_x0 == 0) {
            printf("Error: Division by zero in the Newton-Raphson method.\n");
            return 1; 
        }
 x1 = x0 - f_x0 / f_prime_x0;

        printf("%d\t\t %.6f\t %.6f\t %.6f\n", iter + 1, x0, x1, f(x1));

        if (fabs(f(x1)) < tol) {
            printf("Root found: %.6f\n", x1);
            return 0; 
        }
        x0 = x1;
        iter++;           
    }
              printf("Method did not converge after %d iterations.\n", max_iter);

}

        {2, -1, 1, 3},
        {1, 2, 3, 8},
        {3, 1, -1, 3}
    };

    for (i = 0; i < N; i++)
       {
        float pivot = A[i][i];
        for (j = 0; j <= N; j++) 
        {
            A[i][j] /= pivot;
        }

        for (k = 0; k < N; k++) 
{
            if (k != i)
 {
                float factor = A[k][i];
                for (j = 0; j <= N; j++)
 {
                    A[k][j] -= factor * A[i][j];
                }
            }
        }
    }
    printf("Solutions:\n");

{
        printf("x%d = %.2f\n", i + 1, A[i][N]);

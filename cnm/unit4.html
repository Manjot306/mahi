 #include< stdio.h >
           double factorial(int n) {
            double result = 1.0;
            for (int i = 1; i <= n; i++) {
                result *= i;
            }
            return result;
        }
        double taylorSeries(double x, int terms) {
            double sum = 0.0;
            for (int n = 0; n < terms; n++) {
                sum += (pow(x, n) / factorial(n));
            }
            return sum;
        }
        int main() {
            double x;
            int terms;
            printf("Enter the value of x: ");
            scanf("%lf", &x);
            printf("Enter the number of terms in the Taylor series: ");
            scanf("%d", &terms);
            double result = taylorSeries(x, terms);
            printf("e^%.2f ≈ %.6f\n", x, result);
            return 0;
        }

        #include < stdio.h >
        #include < math.h >
        double chebyshev(int n, double x) {
            if (n == 0) {
                return 1;
            } else if (n == 1) {
                return x;
            } else {
                return 2 * x * chebyshev(n - 1, x) - chebyshev(n - 2, x);
            }
        }
         int main() {
            int n;
            double x;
            printf("Enter the degree of the Chebyshev polynomial (n): ");
            scanf("%d", &n);
            printf("Enter the value of x: ");
            scanf("%lf", &x);
            double result = chebyshev(n, x);
            printf("T_%d(%.2f) = %.6f\n", n, x, result);
            return 0;
        }

        double f(double t, double y) {

        void rungeKutta2(double t0, double y0, double h, int n) {
            double t = t0;
            double y = y0;
            
            printf("t0 = %.2f, y0 = %.2f\n", t0, y0);
            printf("Using step size h = %.2f\n", h);
            printf("Results:\n");
            printf("t\t\ty\n");
            
            for (int i = 0; i <= n; i++) {
                printf("%.2f\t%.5f\n", t, y);
                
                double k1 = h * f(t, y);
                double k2 = h * f(t + h, y + k1);
                
                y += (k1 + k2) / 2;
                t += h;
            }
        }
        
        int main() {
            double t0 = 0.0;
            double y0 = 1.0;
            double h = 0.1;
            int n = 10;
            
            rungeKutta2(t0, y0, h, n);
            
            return 0;
        }

         #include< stdio.h >

        double f(double t, double y) {
            return t + y;  // Example: dy/dt = t + y
        }
        
        void rungeKutta4(double t0, double y0, double h, int n) {
            double t = t0;
            double y = y0;
        
            printf("t0 = %.2f, y0 = %.2f\n", t0, y0);
            printf("Using step size h = %.2f\n", h);
            printf("Results:\n");
            printf("t\t\ty\n");
        
            for (int i = 0; i <= n; i++) {
                printf("%.2f\t%.5f\n", t, y);
        
                double k1 = h * f(t, y);
                double k2 = h * f(t + h / 2, y + k1 / 2);
                double k3 = h * f(t + h / 2, y + k2 / 2);
                double k4 = h * f(t + h, y + k3);
        
                y += (k1 + 2 * k2 + 2 * k3 + k4) / 6;
                t += h;
            }
        }
        
        int main() {
            double t0 = 0.0;
            double y0 = 1.0;
            double h = 0.1;
            int n = 10;
        
            rungeKutta4(t0, y0, h, n);
        
            return 0;
        }

         void modified_euler(float x0, float y0, float h, int n) {
            float x = x0;
            float y = y0;
            printf("x\t\ty\n");
            for (int i = 0; i <= n; i++) {
                printf("%.2f\t\t%.6f\n", x, y);
                float y_predict = y + h * f(x, y);
                y = y + (h / 2) * (f(x, y) + f(x + h, y_predict));
                x = x + h;
            }
        }
        int main() {
            float x0, y0, h;
            int n;
            printf("Enter initial values (x0, y0): ");
            scanf("%f %f", &x0, &y0);
            printf("Enter step size (h): ");
            scanf("%f", &h);
            printf("Enter number of steps (n): ");
            scanf("%d", &n);
        
            modified_euler(x0, y0, h, n);
        
            return 0;
        }
